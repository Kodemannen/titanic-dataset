\begin{equation}

    \begin{split}
        p(A,B) & = p(A| B)p(B) \\
        p(B,A) & = p(B|A)p(A)
    \end{split}
\end{equation}

and
\begin{equation}
    p(A,B) = p(B,A)
\end{equation}
meaning 
\begin{equation}
    p(A|B) =  \frac{p(B|A)p(A)}{p(B)}
\end{equation}
